#include <iostream>
#include <time.h>
#include <sstream>
#include <string>
#include <windows.h>
#include <stdlib.h>
using namespace std;
 
 
int main() {
    HANDLE consoleHandle = GetStdHandle(STD_OUTPUT_HANDLE);
    COORD buffersize = { 160, 80 };
    SMALL_RECT windowsize = { 0,0,159,79 };
 
    SetConsoleScreenBufferSize(consoleHandle, buffersize);
    SetConsoleWindowInfo(consoleHandle, TRUE, &windowsize);
 
    SetConsoleTitle(L"Base Ball");
 
    srand((unsigned int)time(0));
    int iNumber[9] = {};
 
    //1~9까지 숫자를 배열에 넣어준다
    for (int i = 0; i < 9; ++i) {
        iNumber[i] = i + 1;
    }
 
    //설정된 숫자를 섞어준다.
    int iTemp = 0, idx1 = 0, idx2 = 0;
 
    for (int i = 0; i < 100; ++i) {
        idx1 = rand() % 9;
        idx2 = rand() % 9;
 
        iTemp = iNumber[idx1];
        iNumber[idx1] = iNumber[idx2];
        iNumber[idx2] = iTemp;
    }
 
 
    int iStrike = 0, iBall = 0;
    int iInput[3] = { 0 };
 
    printf("==================================================================================\n");
    printf("숫자 야구 게임에 오신 여러분 환영합니다!\n");
    printf("숫자 3개를 중복없이 입력해주시면 랜덤으로 만들어진 숫자와 비교하여 알려드립니다!\n");
    printf("숫자와 자릿수를 모두 맞히면 Strike, 숫자만 맞으면 Ball이 출력됩니다!!\n");
    printf("숫자는 1 ~ 9까지의 정수를 입력해주셔야합니다!\n");
    printf("그리고 0을 입력하면 게임이 종료되니 조심하세요!\n");
    printf("==================================================================================\n");
 
    while (true) {
 
        cout << "숫자 3개를 입력해주세요! (입력 예시 : 1 2 3) : ";
 
        // 사용자의 입력을 받아들입니다.
        string inputLine;
        getline(cin, inputLine);
 
        // 입력이 '0'일 경우, 게임을 종료합니다.
        if (inputLine[0] == '0')
            break;
 
        // 사용자가 입력한 문자열을 스트링 스트림으로 변환합니다.
        istringstream ss(inputLine);
        // 숫자 입력 카운트 변수를 초기화합니다.
        int count = 0;
 
        // 최대 3개의 숫자를 입력받습니다.
        while (count < 3 && ss >> iInput[count]) {
            count++;
            // 두 번째 이후의 숫자 사이의 쉼표를 무시합니다.
            if (count < 3 && (ss.peek() == ' '))
                ss.ignore();
        }
 
        // 추가적인 숫자가 입력되었을 경우 또는 3개 미만의 숫자가 입력된 경우
        if (count < 3 || count >= 4 || (count == 3 && (ss >> iInput[0])))
        {
            cout << "3개의 숫자를 입력해주세요." << endl;
            continue;
        }
 
        //1보다 작거나 9보다 큰 수가 입력되었을 경우
        if (iInput[0] < 0 || iInput[0] > 9
            || iInput[1] < 0 || iInput[1] > 9 || iInput[2] < 0 || iInput[2] > 9)
        {
            cout << "잘못된 숫자를 입력하였습니다." << endl;
            continue;
        }
 
        // 세 개의 숫자 중에 중복이 생기는 경우
        else if (iInput[0] == iInput[1] || iInput[1] == iInput[2] || iInput[2] == iInput[0])
        {
            cout << "서로 다른 세 개의 숫자를 입력해주세요." << endl;
            continue;
        }
 
        iStrike = iBall = 0;
 
        for (int i = 0; i < 3; ++i)
        {
            for (int j = 0; j < 3; ++j)
            {
                //i는 랜덤으로 설정해 둔 내가 맞춰야 할 숫자, j는 내가 입력한 숫자
                if (iNumber[i] == iInput[j])
                {
                    // 내가 입력해둔 숫자와 하나씩 맞춰보고, 맞은 경우 몇 번째에 입력했는지 파악
                    // 자릿수가 같다면 strike의 수가 1 증가
                    if (i == j)
                        ++iStrike;
                    // 자릿수가 다르다면 ball의 수가 1 증가
                    else
                        ++iBall;
                    break;
                }
            }
        }
 
        if (iStrike == 3) {
            cout << "축하합니다! 숫자를 모두 맞췄습니다!" << endl;
            break;
        }
        else if (iStrike == 0 && iBall == 0)
            cout << "Out" << endl;
        else
            cout << iStrike << " Strike " << iBall << " Ball" << endl;
 
    }
    return 0;
 
}
