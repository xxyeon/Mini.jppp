#include <iostream>
#include <vector>
#include <cctype> // 문자열 검사 함수를 사용하기 위한 헤더 추가
#include <Windows.h>
#include <algorithm>
#undef max
 
std::vector<std::vector<char>> board(3, std::vector<char>(3, ' ')); // 3x3 크기의 틱택토 보드를 초기화
 
char currentPlayer = 'X'; // 현재 플레이어를 'X'로 초기화
 
void DrawBoard() {
    std::cout << "  1 2 3" << std::endl; // 열 머리글 출력
    for (int i = 0; i < 3; i++) {
        std::cout << i + 1 << " "; // 행 머리글 출력
        for (int j = 0; j < 3; j++) {
            std::cout << board[i][j]; // 틱택토 보드 내용 출력
            if (j < 2) std::cout << "|"; // 열 구분자 출력
        }
        std::cout << std::endl;
        if (i < 2) std::cout << "  -----" << std::endl; // 행 구분자 출력
    }
}
 
bool PlaceMark(int row, int col) {
    if (row < 0 || row >= 3 || col < 0 || col >= 3 || board[row][col] != ' ') {
        return false; // 잘못된 위치 또는 이미 기호가 놓인 위치일 경우 실패
    }
    board[row][col] = currentPlayer; // 유효한 위치에 기호를 놓음
    return true;
}
 
void SwitchPlayer() {
    if (currentPlayer == 'X') {
        currentPlayer = 'O'; // 현재 차례를 교체 (X -> O)
    }
    else {
        currentPlayer = 'X'; // 현재 차례를 교체 (O -> X)
    }
}
 
bool CheckWin() {
    // 가로, 세로, 대각선 방향으로 승리 확인
    for (int i = 0; i < 3; i++) {
        if (board[i][0] == currentPlayer && board[i][1] == currentPlayer && board[i][2] == currentPlayer) {
            return true; // 가로로 3개 연속 기호가 같으면 승리
        }
        if (board[0][i] == currentPlayer && board[1][i] == currentPlayer && board[2][i] == currentPlayer) {
            return true; // 세로로 3개 연속 기호가 같으면 승리
        }
    }
    if (board[0][0] == currentPlayer && board[1][1] == currentPlayer && board[2][2] == currentPlayer) {
        return true; // 왼쪽 상단에서 오른쪽 하단 대각선으로 3개 연속 기호가 같으면 승리
    }
    if (board[0][2] == currentPlayer && board[1][1] == currentPlayer && board[2][0] == currentPlayer) {
        return true; // 왼쪽 하단에서 오른쪽 상단 대각선으로 3개 연속 기호가 같으면 승리
    }
    return false;
}
 
bool CheckTie() {
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            if (board[i][j] == ' ') {
                return false; // 아직 놓을 공간이 남아있으면 게임은 무승부가 아님
            }
        }
    }
    return true; // 모든 공간이 채워져 있으면 게임은 무승부
}
 
int main() {
    HANDLE consoleHandle = GetStdHandle(STD_OUTPUT_HANDLE);
    COORD buffersize = { 160, 80 };
    SMALL_RECT windowsize = { 0,0,159,79 };
 
    SetConsoleScreenBufferSize(consoleHandle, buffersize);
    SetConsoleWindowInfo(consoleHandle, TRUE, &windowsize);
 
    SetConsoleTitle(L"Tic Tac Toe");
 
    int row, col = 0;
 
    std::cout << "틱택토 게임을 시작합니다!" << std::endl;
    
    while (true) {
        DrawBoard();
        std::cout << "플레이어 " << currentPlayer << "의 차례입니다. 행과 열을 숫자로 입력하세요 (예: 1, 2): ";
        scanf_s("%d, %d", &row, &col);
 
        // 숫자 입력 검사
        if (std::cin.fail()) {
            std::cin.clear();
            std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
            std::cout << "숫자를 입력하세요. 다시 시도하세요." << std::endl;
            continue;
        }
 
        if (row < 1 || row > 3 || col < 1 || col > 3) {
            std::cout << "잘못된 위치입니다. 다시 시도하세요." << std::endl;
            continue;
        }
 
        row--; // 0부터 시작하는 인덱스로 변환
        col--;
 
        if (PlaceMark(row, col)) {
            if (CheckWin()) {
                DrawBoard();
                std::cout << "플레이어 " << currentPlayer << "이(가) 이겼습니다! 축하합니다!" << std::endl;
                break;
            }
            else if (CheckTie()) {
                DrawBoard();
                std::cout << "무승부입니다. 게임 종료!" << std::endl;
                break;
            }
            SwitchPlayer();
        }
        else {
            std::cout << "이미 놓인 위치입니다. 다시 시도하세요." << std::endl;
        }
    }
 
    return 0;
}
