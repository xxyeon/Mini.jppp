#include <iostream>
#include <string>
#include <ctime>
#include <cstdlib>
#include <list>
#include <vector>
#include <windows.h>
using namespace std;
 
int main() {
    HANDLE consoleHandle = GetStdHandle(STD_OUTPUT_HANDLE);
COORD buffersize = { 160, 80 };
SMALL_RECT windowsize = { 0,0,159,79 };
 
SetConsoleScreenBufferSize(consoleHandle, buffersize);
SetConsoleWindowInfo(consoleHandle, TRUE, &windowsize);
 
SetConsoleTitle(L"Hang Man");
 
    srand(time(NULL));
    int lst = 0;
    char ch;
    
    cout << "리스트에 넣을 단어의 수를 입력해 주세요: ";
    cin >> lst;
 
    vector<string> answers;
 
    for (int i = 0; i < lst; i++) {
        string word;
        cout << (i + 1) << "번째 단어를 입력해 주세요: ";
        cin >> word;
        answers.push_back(word);
    }
    
    int r = rand() % (lst + 1);
    string str = answers[r];
    cout << "문자를 맞춰보세요 : ";
    string res(str.length(), '_');
    cout << res << endl;
 
    while (true) {
        cout << "문자를 입력하세요 : ";
        cin >> ch;
 
        for (int j = 0; j < str.length(); j++) {
            if (ch == str[j]) {
                res[j] = ch;
                cout << res << endl;
            }
        }
        if (str == res) {
            cout << "성공" << endl;
            break;
        }
    }
 
    return 0;
}
