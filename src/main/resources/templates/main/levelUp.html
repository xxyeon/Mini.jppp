#include <iostream>
#include <cstdlib>
#include <ctime>
#include <stdlib.h>
#include <Windows.h>
#undef min
#undef max
#include <algorithm>
using namespace std;
 
 
class LevelUpGame {
private:
    int level;
    int power;
 
public:
    LevelUpGame() : level(1), power(50) {}
 
    void WelcomeMessage() {
        cout << "======================================================" << endl;
        cout << "반가워 ! 용사야 !! 최강 용사가 되기 위해 온 걸 환영해!" << endl;
        cout << "지금 부터 다른 용사와 너는 최대 3번의 원펀치 대결을 하게 될 거야!" << endl;
        cout << "너는 힘 50과 레벨 1을 가지고 시작해!" << endl;
        cout << "너는 한 판에 0부터 30 사이의 힘을 쓸 수 있고" << endl;
        cout << "상대보다 많은 힘을 쓰면 한 판을 이기게 돼!" << endl;
        cout << "게임에서 이기면 레벨이 2 상승하고 지면 -1만큼 감소돼" << endl;
        cout << "레벨 4를 달성하면 너가 최강 용사가 될 수 있어!" << endl;
        cout << "너가 사용한 힘은 다시 재사용이 안되니 유의해서 사용해줘!" << endl;
        cout << "그럼 행운을 빌게 용사야!" << endl;
        cout << "=======================================================" << endl;
    }
 
    void startGame() {
        for (int i = 0; i < 3; ++i) {
            std::cout << "========== 현재 " << i + 1 << " 단계 ==========" << std::endl;
            std::cout << "너의 현재 레벨: " << level << std::endl;
            std::cout << "너의 현재 힘: " << power << std::endl;
 
            printf("   ^  ^                  ^  ^ \n");
            printf("  ( O_O)                (O_O )\n");
            printf("   ( - )-              -( - )\n");
            printf("   /   \\                /   \\\n");
 
            int computerPower = rand() % 31; // 컴퓨터의 랜덤한 힘 결정 (0부터 30까지)
            int yourPower;
            std::cout << "힘을 입력해줘 용사야!: ";
            std::cin >> yourPower;
 
            yourPower = std::min(std::max(yourPower, 0), 30); // 입력 범위를 0에서 30으로 제한
            std::cout << "적의 힘은 " << computerPower << std::endl;
 
            if (yourPower >= computerPower) {
                std::cout << "네가 이겼어 용사야!" << std::endl;
                level += 2;
            }
            else {
                std::cout << "아쉽지만 다음 판에서 힘내보자!" << std::endl;
                level -= 1;
            }
 
            power -= yourPower; // 사용한 힘만큼 전투 후 힘 차감
 
            if (level >= 4) {
                std::cout << "축하해! 넌 이제 최강용사가 되었어!!" << std::endl;
                return;
            }
        }
        std::cout << "아직 최강용사가 되기엔 수련이 더 필요해! 안녕~" << std::endl;
 
    }
};
 
int main() {
 
 
    HANDLE consoleHandle = GetStdHandle(STD_OUTPUT_HANDLE);
    COORD buffersize = { 160, 80 };
    SMALL_RECT windowsize = { 0,0,159,79 };
 
    SetConsoleScreenBufferSize(consoleHandle, buffersize);
    SetConsoleWindowInfo(consoleHandle, TRUE, &windowsize);
 
    SetConsoleTitle(L"Level Up");
    srand(static_cast<unsigned int>(time(nullptr)));
 
    LevelUpGame game;
    string choice;
 
    game.WelcomeMessage();
 
    do {
        cout << "게임을 시작할까요? (y/n): ";
        cin >> choice;
 
        if (choice != "y" && choice != "n") {
            cout << "잘못된 입력입니다. 다시 입력해주세요." << endl;
        }
        else if (choice == "y") {
            system("cls");
            game.startGame();
            break;
        }
        else if (choice == "n") {
            cout << "게임을 종료합니다." << endl;
            break;
        }
    } while (true);
 
    return 0;
}
